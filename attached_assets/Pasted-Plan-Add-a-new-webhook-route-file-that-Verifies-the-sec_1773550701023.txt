A. Add src/webhook.js
Create file src/webhook.js with this exact content:

const express = require('express');
const router = express.Router();

// Simple defensive helper: parse bigints if you use them. Adjust if unnecessary.
function safeJson(obj) {
try { return JSON.stringify(obj); } catch (e) { return String(obj); }
}

// POST /webhook/:secret
router.post('/webhook/:secret', async (req, res) => {
try {
const secret = req.params.secret;
if (!process.env.TELEGRAM_WEBHOOK_SECRET) {
console.error('TELEGRAM_WEBHOOK_SECRET not set in env');
return res.sendStatus(500);
}
if (secret !== process.env.TELEGRAM_WEBHOOK_SECRET) {
console.warn('Invalid webhook secret attempt');
return res.sendStatus(403);
}

const update = req.body;
// Quick debug log (update_id only)
console.log('TG update received:', { update_id: update && update.update_id });

// Minimal DB write: insert into messages table (adjust table/columns to your schema)
// This assumes you use node-postgres (pg) and exported a pool from ../db or process.env.DATABASE_URL
try {
  const { pool } = require('./db'); // adjust path if needed; see note below
  const now = new Date().toISOString();
  const payload = safeJson(update);
  // Example table: messages (source_chat_id BIGINT, source_message_id BIGINT, payload JSONB, status TEXT, created_at TIMESTAMP)
  await pool.query(
    `INSERT INTO messages (source_chat_id, source_message_id, payload, status, created_at)
     VALUES ($1, $2, $3::jsonb, $4, $5)`,
    [
      (update.message && update.message.chat && update.message.chat.id) || null,
      (update.message && update.message.message_id) || null,
      payload,
      'received',
      now,
    ]
  );
} catch (dbErr) {
  // Log DB error but do NOT block response
  console.error('DB insert failed (continuing):', dbErr && dbErr.message);
}

// Respond quickly
return res.sendStatus(200);
} catch (err) {
console.error('Unhandled error in webhook handler:', err && err.stack ? err.stack : err);
// Return 200 if you want Telegram to stop retrying even if internal error
return res.sendStatus(500);
}
});

module.exports = router;

Notes about db import:

I referenced require('./db') expecting a db module that exports { pool }, using node-postgres (pg). If your repo already has a DB helper, adjust the require path. If you do NOT have one, add the minimal db file below.

Troubleshooting quick guide (if something fails)

403 from Telegram (after setWebhook) or 403 in logs: webhook path's secret doesn't match TELEGRAM_WEBHOOK_SECRET env. Confirm exact match (no spaces/newlines).
No logs: app not deployed or wrong domain. Confirm Railway build succeeded and public domain is live.
DB insert errors: inspect error in logs; confirm DATABASE_URL is set and reachable from Railway service; run psql externally to validate credentials if needed.
Telegram last_error says 502 or timeout: your handler returned non-200 or took too long. Make sure the route returns 200 immediately.
